Sub ClearNonPrintArea()
    Dim ws As Worksheet
    Dim printAreaRange As Range
    Dim targetRange As Range
    Dim nonPrintRange As Range

    ' Set the worksheet where you want to clear non-print area content
    Set ws = ThisWorkbook.Sheets("Sheet1") ' Adjust to your worksheet name

    ' Define the specific range that contains your data (e.g., A1:D100)
    Set targetRange = ws.Range("A1:D100") ' Adjust according to your data range

    ' Ensure the print area is set and fetch the print area range
    If ws.PageSetup.PrintArea <> "" Then
        Set printAreaRange = ws.Range(ws.PageSetup.PrintArea)
    Else
        MsgBox "No print area set in the specified worksheet."
        Exit Sub
    End If

    ' Identify cells outside the print area by subtracting the print area from the target range
    On Error Resume Next ' Prevents errors if no cells are found
    Set nonPrintRange = targetRange.SpecialCells(xlCellTypeConstants + xlCellTypeFormulas)
    If Not nonPrintRange Is Nothing Then
        ' Create a range excluding the print area
        Set nonPrintRange = Application.Intersect(targetRange, nonPrintRange)
        Set nonPrintRange = Application.Subtract(nonPrintRange, printAreaRange)

        ' Clear all cells outside the print area within the target range
        nonPrintRange.Clear
    End If
    On Error GoTo 0 ' Restore error handling

    MsgBox "All cells outside the print area within the target range have been cleared."
End Sub
